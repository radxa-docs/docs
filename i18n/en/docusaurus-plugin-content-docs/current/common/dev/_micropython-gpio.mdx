import machine
import time

class Blinker:

def __init__(self, pin_num, interval):
        self.pin = machine.Pin(pin_num, machine.Pin.OUT)
        self.interval = interval
        self.pin.off()

    def blink(self):
        self.pin.toggle()
        time.sleep(self.interval)

def main():
    blinker = Blinker(#gpio#, #time#)

    try:
        while True:
            blinker.blink()
    except KeyboardInterrupt:
        print("Blink stopped.")

if __name__ == "__main__":
    main()
